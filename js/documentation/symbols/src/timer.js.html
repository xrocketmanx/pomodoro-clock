1 /**
  2  * Sets timer that will call action every
  3  * second and give it current seconds and
  4  * in the end calls callback;
  5  * has time converting static methods
  6  * @class
  7  * @param {Number}   seconds  timeout in seconds
  8  * @param {Function} action   actions every second
  9  * @param {Function} callback actions in the end
 10  */
 11 function Timer(seconds, action, callback) {
 12 	var interval;
 13 	var initialSeconds = seconds;
 14 
 15 	/**
 16 	 * Starts timer or restarts if arguments are ommited
 17 	 * (all parameters are optional)
 18 	 * @param  {Number} _seconds  timeout in seconds
 19 	 * @param  {Function} _action   action every second
 20 	 * @param  {Function} _callback actions in the end
 21 	 */
 22 	this.start = function(_seconds, _action, _callback) {
 23 		if (_seconds) {
 24 			initialSeconds = _seconds;
 25 		}
 26 		if (_action) {
 27 			action = _action;
 28 		}
 29 		if (_callback) {
 30 			callback = _callback;
 31 		}
 32 		seconds = initialSeconds;
 33 		start();
 34 	};
 35 
 36 	/**
 37 	 * Continues timer execution
 38 	 */
 39 	this.continue = function() {
 40 		start();
 41 	};
 42 
 43 	/**
 44 	 * Stops execution of timer
 45 	 */
 46 	this.stop = function() {
 47 		if (interval) {
 48 			clearInterval(interval);
 49 		}
 50 	};
 51 
 52 	/**
 53 	 * Returns seconds of current Timer
 54 	 * state
 55 	 * @return {Number} seconds
 56 	 */
 57 	this.getSeconds = function() {
 58 		return seconds;
 59 	}
 60 
 61 	/**
 62 	 * Sets timeout in seconds
 63 	 * @param {Number} _seconds timeout
 64 	 */
 65 	this.setTimeout = function(_seconds) {
 66 		initialSeconds = _seconds;
 67 	}
 68 
 69 	function start() {
 70 		action(seconds);
 71 		interval = setInterval(function() {
 72 			action(--seconds);
 73 			if (seconds === 0) {
 74 				clearInterval(interval);
 75 				if (callback) callback();
 76 				return;
 77 			}
 78 		}, 1000);
 79 	};
 80 }
 81 
 82 /**
 83  * Convert time to seconds
 84  * @param  {Number} hours   
 85  * @param  {Number} minutes 
 86  * @param  {Number} seconds 
 87  * @return {Number}         time in seconds
 88  */
 89 Timer.convertToSeconds = function(hours, minutes, seconds) {
 90 	return hours * 3600 + minutes * 60 + seconds;
 91 };
 92 
 93 /**
 94  * Gets seconds from time in seconds
 95  * @param  {Number} seconds time in seconds
 96  * @return {Number}         seconds
 97  */
 98 Timer.getSecondsFromTime = function(seconds) {
 99 	return seconds % 60;
100 };
101 
102 /**
103  * Gets minutes from time in seconds
104  * @param  {Number} minutes time in seconds
105  * @return {Number}         minutes
106  */
107 Timer.getMinutesFromTime = function(seconds) {
108 	return Math.floor(seconds / 60) % 60;
109 };
110 
111 /**
112  * Gets hours from time in seconds
113  * @param  {Number} hours time in seconds
114  * @return {Number}       hours
115  */
116 Timer.getHoursFromTime = function(seconds) {
117 	return Math.floor(seconds / 3600);
118 };
