1 /**
  2  * Has workingTime and freeTime timers that
  3  * working one after another in loop
  4  * @class
  5  * @param {Number}   working         working timeout seconds
  6  * @param {Number}   free            free timeout seconds
  7  * @param {Function} workingAction   actions every seconds of working time
  8  * @param {Function} freeAction      actions every seconds of free time
  9  * @param {Function} workingCallback actions after working time
 10  * @param {Function} freeCallback    actions after free time
 11  */
 12 function PomodoroTimer(working, free, workingAction, freeAction, workingCallback, freeCallback) {
 13 	var workingTimer = new Timer(working, workingAction, function() {
 14 		workingCallback();
 15 		freeTimer.start();
 16 	});
 17 	var freeTimer = new Timer(free, freeAction, function() {
 18 		freeCallback();
 19 		workingTimer.start();
 20 	}); 
 21 
 22 	/**
 23 	 * Starts timer with initial parameters or
 24 	 * with function params if they are not ommited
 25 	 * (all parameters are optional)
 26 	 * @param  {Number} working working timeout seconds
 27 	 * @param  {Number} free    free timeout seconds
 28 	 */
 29 	this.start = function(working, free) {
 30 		if (working && free) {
 31 			workingTimer.setTimeout(working);
 32 			freeTimer.setTimeout(free);
 33 		}
 34 
 35 		workingTimer.start();
 36 	};
 37 
 38 	/**
 39 	 * Continues working of timer
 40 	 */
 41 	this.continue = function() {
 42 		if (workingTimer.getSeconds()) {
 43 			workingTimer.continue();
 44 		} else {
 45 			freeTimer.continue();
 46 		}
 47 	};
 48 
 49 	/**
 50 	 * Stops timer
 51 	 */
 52 	this.stop = function() {
 53 		workingTimer.stop();
 54 		freeTimer.stop();
 55 	};
 56 }
